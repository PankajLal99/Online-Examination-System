Fullscreen with JavaScript

Click on the "Open Fullscreen" button to open this page in fullscreen mode. Close it by either clicking the "Esc" key on your keyboard, or with the "Close Fullscreen" button.

var ccount = 0; var acount = 0; document.addEventListener("keydown", function (event) { if (event.altKey) { // document.body.innerHTML = `   You have pressed alt+tab `; alert("alt pressed") acount++; console.log(acount); } if (event.ctrlKey) { // document.body.innerHTML = `   You have pressed alt+tab `; alert("ctrl pressed") ccount++; console.log(ccount); } } ); var elem = document.getElementById("quiz-screen"); function openFullscreen() { if (elem.requestFullscreen) { elem.requestFullscreen(); } else if (elem.webkitRequestFullscreen) { /* Safari */ elem.webkitRequestFullscreen(); } else if (elem.msRequestFullscreen) { /* IE11 */ elem.msRequestFullscreen(); } } function closeFullscreen() { if (document.exitFullscreen) { document.exitFullscreen(); } else if (document.webkitExitFullscreen) { /* Safari */ document.webkitExitFullscreen(); } else if (document.msExitFullscreen) { /* IE11 */ document.msExitFullscreen(); } }
